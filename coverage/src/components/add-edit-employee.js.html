  async handleSave(e) {

    console.log(employeeData)
    const errors = validate(employeeData , this.validationRules);

    // Inform the user

      text: this.id

    if(this.id) {

      store.dispatch(addEmployee(employeeData ));

    // Success message
    await Swal.fire({

      icon: 'success',

  }
